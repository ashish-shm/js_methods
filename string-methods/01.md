### abbrev_name
Write a JavaScript function *to convert* a string in *abbreviated form*.

Example:
```js
  // your code goes here
  function abbrev_name(str){
    let newstr = str.split(" ");
    let x = "";
    for(let i = 0 ; i < newstr.length ; i++){
      if ( i == 0){
        x = x + newstr;
      }
      else
      x = " " + newstr[i].slice(0,1);
    }
    return x;
  }
  console.log(abbrev_name("Robin Singh"));
  "Robin S."
```
